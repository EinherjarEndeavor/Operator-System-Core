import cv2
import numpy as np

def enhance_image(input_path, output_path):
    # Load image
    img = cv2.imread(input_path)
    
    # Apply super-resolution
    sr = cv2.dnn_superres.DnnSuperResImpl_create()
    sr.readModel('EDSR_x4.pb')  # Download this model
    sr.setModel('edsr', 4)
    result = sr.upsample(img)
    
    # Denoise
    denoised = cv2.fastNlMeansDenoisingColored(result, None, 10, 10, 7, 21)
    
    # Save enhanced image
    cv2.imwrite(output_path, denoised)

import pickle

def extract_face_features(image_path):
    # Load image
    image = face_recognition.load_image_file(image_path)
    
    # Find face locations
    face_locations = face_recognition.face_locations(image, model="cnn")
    
    # Create face encodings
    face_encodings = face_recognition.face_encodings(image, face_locations)
    
    return face_encodings

Step 4: Matching System

python
def find_matches(unknown_encoding, database_path="face_database.pkl"):
    with open(database_path, "rb") as f:
        database = pickle.load(f)
    
    matches = []
    for name, encoding in database.items():
        distance = face_recognition.face_distance([encoding], unknown_encoding)[0]
        if distance < 0.6:  # Threshold for matching
            matches.append((name, distance))
    
    return sorted(matches, key=lambda x: x[1])
